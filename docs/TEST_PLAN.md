### Playwright 测试

```typescript
// tests/gameplay.spec.ts
import { test, expect } from '@playwright/test'

test.describe('单人游戏功能', () => {
  test('玩家移动测试', async ({ page }) => {
    await page.goto('http://localhost:3001')
    await page.waitForSelector('#game-container')
    
    // 聚焦 Canvas
    const canvas = page.locator('canvas')
    await canvas.focus()
    await page.waitForTimeout(2000)
    
    // 右键点击移动
    await canvas.click({ button: 'right', position: { x: 400, y: 300 } })
    await page.waitForTimeout(1000)
    
    // 检查玩家位置是否变化
    const debugOverlay = page.locator('.debug-overlay')
    await expect(debugOverlay).toContainText('玩家位置')
  })
  
  test('玩家攻击测试', async ({ page }) => {
    await page.goto('http://localhost:3001')
    await page.waitForSelector('#game-container')
    
    const canvas = page.locator('canvas')
    await canvas.focus()
    await page.waitForTimeout(2000)
    
    // 左键攻击
    await canvas.click({ button: 'left' })
    await page.waitForTimeout(500)
    
    // 检查攻击特效
    const effects = page.locator('.attack-effect')
    await expect(effects).toHaveCount(1)
  })
})
```
