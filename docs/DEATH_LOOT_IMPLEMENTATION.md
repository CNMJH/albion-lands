### 掉落计算逻辑 ✅

```typescript
if (安全等级 >= 6) {
  掉落 = []  // 安全区不掉落
} else if (安全等级 >= 3) {
  掉落 = 从背包随机选择非装备物品 (0-1 件)
} else if (安全等级 >= 1) {
  掉落 = 从装备栏随机选择装备 (1-2 件)
} else {
  掉落 = 从装备栏随机选择装备 (3-5 件)
}
```
